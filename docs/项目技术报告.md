### 1. 数据处理流水线

#### 1.1 大规模数据采样系统
针对29.3GB原始CSV文件，设计了分层流式采样算法：

```python
import pandas as pd
from sklearn.model_selection import train_test_split

def stratified_large_scale_sampling(input_path, sample_sizes=[10, 100, 1000, 5000]):
    """分层流式采样，保持原始数据分布"""
    sampling_results = {}
    
    for target_size in sample_sizes:
        # 使用chunksize流式处理
        chunks = pd.read_csv(input_path, chunksize=10000)
        sampled_chunks = []
        
        for chunk in chunks:
            if len(sampled_chunks) * 10000 >= target_size:
                break
            
            # 分层采样保持类别比例
            sampled = chunk.groupby('type', group_keys=False).apply(
                lambda x: x.sample(frac=min(1, target_size/len(chunk))))
            sampled_chunks.append(sampled)
        
        result = pd.concat(sampled_chunks)
        sampling_results[target_size] = result
    
    return sampling_results
```

**采样成果**：
- ✅ **测试集**：10条数据 - 代码流程验证
- ✅ **微型集**：100条数据 - 快速原型开发
- ✅ **小型集**：1000条数据 - ⭐主要实验数据集
- ✅ **中型集**：5000条数据 - 稳定性验证

#### 1.2 文本预处理流程
```python
import re
import html

def text_preprocessing_pipeline(text):
    """完整的文本预处理流程"""
    if pd.isna(text):
        return ""
    
    # 1. HTML解码
    text = html.unescape(text)
    
    # 2. 移除HTML标签
    text = re.sub(r'<[^>]+>', '', text)
    
    # 3. 处理特殊字符和空白
    text = re.sub(r'\s+', ' ', text)
    text = text.strip()
    
    # 4. 编码统一（UTF-8）
    text = text.encode('utf-8', errors='ignore').decode('utf-8')
    
    return text
```

```python
models = {

        n_estimators=100, max_depth=None, random_state=42
    ),
    'svm': SVC(kernel='linear', C=1.0, probability=True),
    'naive_bayes': MultinomialNB(alpha=1.0)
}

# 交叉验证评估
def evaluate_models(X, y, models_dict, cv=5):
    """5折交叉验证评估多个模型"""
    results = {}
    
    for name, model in models_dict.items():
        scores = cross_val_score(model, X, y, cv=cv, 
                                scoring='accuracy', n_jobs=-1)
        results[name] = {
            'mean_accuracy': scores.mean(),
            'std_accuracy': scores.std(),
            'scores': scores.tolist()
        }
    
    return results
```

### 4. 实验管理系统

#### 4.1 统一实验接口
```python
import json
from datetime import datetime
import joblib

class ExperimentManager:
    """统一实验管理框架"""
    
    def __init__(self, experiment_name):
        self.name = experiment_name
        self.timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        self.results = {}
        
    def run_experiment(self, data_loader, feature_extractor, model):
        """执行完整实验流程"""
        # 1. 数据加载
        data = data_loader.load()
        
        # 2. 特征提取
        X, y = feature_extractor.transform(data)
        
        # 3. 模型训练与评估
        start_time = datetime.now()
        cv_scores = cross_val_score(model, X, y, cv=5)
        training_time = (datetime.now() - start_time).total_seconds()
        
        # 4. 结果记录
        self.results = {
            'experiment_name': self.name,
            'timestamp': self.timestamp,
            'data_info': data_loader.get_info(),
            'feature_dim': X.shape[1],
            'mean_accuracy': cv_scores.mean(),
            'std_accuracy': cv_scores.std(),
            'training_time_seconds': training_time,
            'cv_scores': cv_scores.tolist()
        }
        
        # 5. 模型保存
        model_path = f"models/{self.name}_{self.timestamp}.joblib"
        joblib.dump(model, model_path)
        
        return self.results
    
    def save_report(self):
        """保存实验报告"""
        report_path = f"reports/{self.name}_{self.timestamp}.json"
        with open(report_path, 'w', encoding='utf-8') as f:
            json.dump(self.results, f, indent=2, ensure_ascii=False)
```

#### 4.2 时间预测系统
```python
def predict_experiment_time(data_size, model_complexity='medium'):
    """基于数据规模和模型复杂度的实验时间预测"""
    
    # 基准时间（分钟）
    time_estimates = {
        'simple': {    # 逻辑回归、朴素贝叶斯
            100: (0.5, 2),
            1000: (2, 5),
            5000: (5, 10),
            10000: (10, 20)
        },
        'medium': {    # SVM、随机森林
            100: (1, 3),
            1000: (3, 8),
            5000: (8, 15),
            10000: (15, 30)
        },
        'complex': {   # 深度学习基准
            100: (5, 15),
            1000: (15, 45),
            5000: (45, 120),
            10000: (120, 240)
        }
    }
    
    # 找到最接近的数据规模
    available_sizes = list(time_estimates[model_complexity].keys())
    closest_size = min(available_sizes, key=lambda x: abs(x - data_size))
    
    min_time, max_time = time_estimates[model_complexity][closest_size]
    
    # 线性缩放
    scale_factor = data_size / closest_size
    estimated_min = min_time * scale_factor
    estimated_max = max_time * scale_factor
    
    return {
        'estimated_minutes': (estimated_min, estimated_max),
        'confidence_interval': '基于历史实验数据的统计估计',
        'recommendation': '建议从较小规模开始验证'
    }
```
